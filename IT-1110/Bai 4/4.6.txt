#include<stdio.h>
#include<math.h>
#include<conio.h>
double x;
double a= 0;
double b= 0;
double c= 0;
double d= 1;
int n;
int main() {
	printf("n = ");scanf("%d", &n);
	printf("x = ");scanf("%lf", &x);
	if(n>0){
		for(double i= 0; i<n+1; i++){
			a= a + pow(x,i);
		}
		for(double i= 0; i<n+1; i++){
			b= b + pow((-1),i)*pow(x,i);
		}
		for(double i= 0; i<n+1; i++){
			for(double k= 1; k<i+1; k++){
			  d= d*k;
		    }
		    c= c+ (pow(x,i)/d);
		}
		printf("a= %lf \n", a);
		printf("b= %lf \n", b);
		printf("c= %lf \n", c);
	}
  return 0;
}
